Create bridge:
yum install bridge-utils

vim /etc/sysconfig/network-scripts/ifcfg-br1
NAME="br1"
DEVICE="br1"
ONBOOT=yes
BOOTPROTO=none
TYPE="BRIDGE"

vim /etc/sysconfig/network-scripts/ifcfg-eno2
NAME="eno2"
DEVICE="eno2"
ONBOOT=yes
NETBOOT=yes
IPV6INIT=no
BOOTPROTO=none
TYPE=Ethernet
BRIDGE=br1

brctl addbr br1
brctl addif br1 eno2

systemctl restart network
